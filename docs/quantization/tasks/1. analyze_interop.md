```python
"""Parse a sim.log produced by SDK 1.4 simulator with simprint markers.

Each marker line has the form: ``@<cycle> PE(<x>,<y>): <prefix>:<tag>``
where prefix ∈ {op_start, op_end, comm_start, comm_end} and tag is the
op/comm name we set in decode.csl / comm_pe.csl.

Returns a DataFrame with columns: pe_x, pe_y, region, occurrence, cycles.
``region`` is e.g. ``op:rmsnorm_x`` or ``comm:all_reduce_bsz``.
``occurrence`` is the 0-indexed dynamic invocation count (since the same
comm primitive is called multiple times per decode step).
"""

import re
import sys
from collections import defaultdict
import pandas as pd

LINE_RE = re.compile(
    r"^@(?P<cycle>\d+)\s+PE\((?P<x>\d+),(?P<y>\d+)\):\s+"
    r"(?P<prefix>op_start|op_end|comm_start|comm_end):(?P<tag>[A-Za-z0-9_]+)"
)


def parse(stream):
    # open[(x,y,kind:tag)] -> list of pending start cycles (FIFO stack)
    open_starts = defaultdict(list)
    rows = []
    occ_count = defaultdict(int)  # (x,y,kind:tag) -> dynamic-invocation index

    for line in stream:
        m = LINE_RE.match(line)
        if not m:
            continue
        cycle = int(m["cycle"])
        x, y = int(m["x"]), int(m["y"])
        prefix, tag = m["prefix"], m["tag"]
        kind = "op" if prefix.startswith("op_") else "comm"
        region = f"{kind}:{tag}"
        key = (x, y, region)
        if prefix.endswith("_start"):
            open_starts[key].append(cycle)
        else:
            if not open_starts[key]:
                # End without start — log corruption or truncation
                continue
            start = open_starts[key].pop(0)
            rows.append({
                "pe_x": x, "pe_y": y, "region": region,
                "occurrence": occ_count[key], "cycles": cycle - start,
            })
            occ_count[key] += 1

    return pd.DataFrame(rows)


if __name__ == "__main__":
    df = parse(sys.stdin if len(sys.argv) < 2 else open(sys.argv[1]))
    df.to_csv(sys.stdout, index=False)
```

## Task 6: Write `compute_breakdown.py` — aggregate to comm/compute/total per config

**Files:**
- Create: `test/decode_interop/analyze/compute_breakdown.py`

```python
"""Aggregate per-PE region cycles into a comm-vs-compute breakdown for one config.

Input:  results/p<P>/sim.log
Output: breakdown_p<P>.csv with columns:
        region, mean_cycles, max_cycles, count_pes, count_invocations
Also prints summary: total e2e mean cycles, comm mean cycles, compute mean cycles, comm/total ratio.

Methodology:
- For each PE, sum cycles across all `op:*` regions => per-PE total e2e cycles for one decode step.
- Mean across PEs => "e2e mean cycles" (matches the launcher's reported mean within ±a few cycles).
- For each PE, sum cycles across all `comm:*` regions (from comm_pe.csl primitives) +
  any `op:reconfig_*` regions (reconfig_allreduce_axis is pure comm, no inner comm tag) =>
  per-PE comm cycles. Mean across PEs => "comm mean cycles".
- compute mean cycles = e2e − comm.
"""

import sys
from pathlib import Path
import pandas as pd
from parse_simlog import parse

COMM_OP_TAGS = {"op:reconfig_x", "op:reconfig_y"}  # purely comm; no inner comm region


def breakdown(simlog_path: Path):
    with simlog_path.open() as f:
        df = parse(f)

    # Per-PE total of all op:* regions = e2e per PE (one decode step)
    op = df[df.region.str.startswith("op:")]
    e2e_per_pe = op.groupby(["pe_x", "pe_y"])["cycles"].sum()

    # Per-PE total of all comm:* regions (from comm_pe.csl) + op:reconfig_*
    comm = df[df.region.str.startswith("comm:") | df.region.isin(COMM_OP_TAGS)]
    comm_per_pe = comm.groupby(["pe_x", "pe_y"])["cycles"].sum()

    e2e_mean = e2e_per_pe.mean()
    comm_mean = comm_per_pe.reindex(e2e_per_pe.index, fill_value=0).mean()
    compute_mean = e2e_mean - comm_mean

    summary = pd.DataFrame([{
        "e2e_mean_cycles": e2e_mean,
        "comm_mean_cycles": comm_mean,
        "compute_mean_cycles": compute_mean,
        "comm_ratio": comm_mean / e2e_mean,
    }])

    per_region = (df.groupby("region")["cycles"]
                    .agg(mean_cycles="mean", max_cycles="max", count_invocations="count")
                    .reset_index())

    return summary, per_region


if __name__ == "__main__":
    simlog = Path(sys.argv[1])
    summary, per_region = breakdown(simlog)
    out_csv = simlog.parent / f"breakdown_{simlog.parent.name}.csv"
    per_region.to_csv(out_csv, index=False)
    print(summary.to_string(index=False))
    print(f"\nPer-region detail written to: {out_csv}")
```

"""Aggregate sweep results into a single CSV + plot of comm overhead vs P."""

import json
from pathlib import Path
import pandas as pd
import matplotlib
matplotlib.use("Agg")
import matplotlib.pyplot as plt
from compute_breakdown import breakdown

    plt.savefig(ROOT / "analyze" / "overhead_vs_P.png", dpi=150)

if __name__ == "__main__":
